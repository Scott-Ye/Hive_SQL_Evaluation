703  def deduplicate_rows(rows: Iterable[Dict[str, str]]) -> Tuple[List[Dict[str, str]], int]:
704      deduped: List[Dict[str, str]] = []
705      seen_sql = set()
706      duplicate_count = 0
707      for row in rows:
708          sql_key = normalize_sql(row["sql_text"])
709          if sql_key in seen_sql:
710              duplicate_count += 1
711              continue
712          seen_sql.add(sql_key)
713          deduped.append(deepcopy(row))
714      return deduped, duplicate_count

717  def quality_check_rows(rows: List[Dict[str, str]]) -> List[str]:
718      issues: List[str] = []
719      for index, row in enumerate(rows, start=1):
720          case_label = row.get("case_id", f"ROW_{index}")
721          if row["gt_status"] not in {"pass", "fail"}:
722              issues.append(f"{case_label}: gt_status 必须是 pass 或 fail")
723          if not row["sql_text"]:
724              issues.append(f"{case_label}: sql_text 不能为空")
725          if row["source_tier"] != "real":
726              issues.append(f"{case_label}: source_tier 必须是 real")
727          if not row["source_ref"]:
728              issues.append(f"{case_label}: source_ref 不能为空")
729          if row["gt_label_strength"] not in ALLOWED_GT_LABEL_STRENGTHS:
730              issues.append(f"{case_label}: gt_label_strength 必须是 strong/medium/weak")
731          if row["baseline_status"] == "fail" and row["baseline_error_position"] and not POSITION_PATTERN.match(
732              row["baseline_error_position"]
733          ):
734              issues.append(f"{case_label}: baseline_error_position 必须是合法的行列格式")
735      return issues
